import { Module } from '@nestjs/common';

import { ConfigModule, ConfigService } from '@nestjs/config';

import { StorageService } from './storage.service';

import { S3Client } from '@aws-sdk/client-s3';




@Module({

  imports: [ConfigModule], // Importa ConfigModule para usar ConfigService

  providers: [

    StorageService,

    {

      provide: 'S3_CLIENT', // Proveedor de cliente S3

      inject: [ConfigService],

      useFactory: (configService: ConfigService): S3Client => {

        // Configuración de AWS S3

        const accessKeyId = configService.get<string>('AWS_ACCESS_KEY_ID');

        const secretAccessKey = configService.get<string>(

          'AWS_SECRET_ACCESS_KEY',

        );

        const region = configService.get<string>('AWS_REGION');




        if (!accessKeyId || !secretAccessKey || !region) {

          throw new Error('AWS credentials are not properly configured');

        }




        return new S3Client({

          region,

          credentials: {

            accessKeyId,

            secretAccessKey,

          },

        });

      },

    },

  ],

  exports: [StorageService], // Exporta el servicio para que otros módulos puedan usarlo

})

export class StorageModule {}
